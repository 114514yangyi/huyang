void swap(int& A, int& B) {
    int temp = A;
    A = B;
    B = temp;
}

int Median_partition(vector<int>& nums, int L, int R) {
    if (R - L + 1 <= 5)
        return rand() % (R - L + 1) + L;

    vector<vector<int>> group;
    for (int i = L; i <= R; i += 5) {
        vector<int> temp;
        for (int j = 0; j < 5; j++) {
            if (i + j <= R)
                temp.push_back(nums[i + j]);
        }
        sort(temp.begin(), temp.end());
        group.push_back(temp);
    }

    vector<int> medians;
    for (vector<int> x : group) {
        if (x.size() >= 3)
            medians.push_back(x[2]);
        else
            medians.push_back(x[x.size() - 1]);
    }

    return selection(medians, medians.size() / 2, 0, medians.size() - 1);
}

int selection(vector<int>& nums, int index, int L, int R) {
    if (nums.size() == 0)
        return -1;
    if (L == R)
        return nums[L];

    int pivot = Median_partition(nums, L, R);
    if (pivot == index)
        return nums[pivot];
    if (pivot > index)
        return selection(nums, index, L, pivot - 1);
    else
        return selection(nums, index, pivot + 1, R);
}
