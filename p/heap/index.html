#include "sample.h"
class Heap {
public:
    int size;
    int* array;
    Heap() {
        size = 0;
        array = new int[100];
        array[0] = -1;
    }

    void insert(int x) {
        size++;
        array[size] = x;
        modify_up(size);
    }

    int getmax() {
        return array[1];
    }

    int extractmax() {
        int res = getmax();
        array[1] = array[size];
        size--;
        modify_down(1);
        return res;
    }

    int* sort() {
        int n = size;
        int* res = new int[n + 1];
        for (int i = 0; i < n; i++) {
            res[i] = extractmax();
        }
        return res;
    }

    void build_heap(const vector<int>& v) {
        size = v.size();
        for (int i = 0; i < size; i++) {
            array[i + 1] = v[i];
        }
        for (int i = (size + 1) / 2; i >= 1; i--) {
            modify_down(i);
        }
    }

    void draw_heap() {
        int level = 0;
        int cnt = 0;
        for (int i = 1; i <= size; i++) {
            if (i == pow(2, level)) {
                cout << endl;
                level++;
                cnt = 0;
            }
            cout << array[i] << " ";
            cnt++;
        }
        cout << endl;
    }

private:
    int parent(int index) {
        return index / 2;
    }

    int left(int index) {
        return index * 2;
    }

    int right(int index) {
        return index * 2 + 1;
    }

    void swap(int index1, int index2) {
        int temp = array[index1];
        array[index1] = array[index2];
        array[index2] = temp;
    }

    void modify_up(int index) {
        if (index == 1)
            return;
        if (array[index] < array[parent(index)])
            return;
        else {
            swap(index, parent(index));
            modify_up(parent(index));
        }
    }

    void modify_down(int index) {
        int id = index;
        id = (array[id] < array[left(index)] && left(index) <= size) ? left(index) : id;
        id = (array[id] < array[right(index)] && right(index) <= size) ? right(index) : id;
        if (id == index)
            return;
        else {
            swap(index, id);
            modify_down(id);
        }
    }
};
