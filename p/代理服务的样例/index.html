import socket

import urllib.parse
import time

def abstract_IP(request_data)-> str:
    print(f"request_data:\n{request_data.decode()}")
    # 解析请求报文
    request_lines = request_data.decode().split("\r\n")
    # 获取请求方法、URL和协议版本
    method, url, protocol = request_lines[0].split() 
    # 解析URL
    url_parts = urllib.parse.urlparse(url)
    hostname = url_parts.netloc
    return hostname

    # 创建与目标服务器的连接
    server_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    server_socket.connect((hostname, 80))
    # 发送客户端请求数据给目标服务器
    server_socket.sendall(request_data)
    # 接收目标服务器响应数据并发送给客户端\
    index=0
    while True:
        print(f"index:{index}")
        response_data = server_socket.recv(1024)
        print(f"response_data:\n{response_data.decode()}")
        time.sleep(1)
        if response_data:
            if client_socket.fileno() == -1:
                print("客户端已关闭连接")
                break
            client_socket.sendall(response_data)

    # 关闭连接
    client_socket.close()
    server_socket.close()

def handle_client(client_socket):
    """
    处理客户端请求
    """
    # 接收客户端请求数据
    request_data = client_socket.recv(1024)
    hostname=abstract_IP(request_data)
    
    # 发送请求数据给目标服务器
    sending_request(request_data,hostname,client_socket)

import socket
import urllib.parse

def send_request(hostname, port, request_data):
    # 创建客户端套接字
    client_socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
    # 连接代理服务器
    client_socket.connect((hostname, port))
    # 发送请求数据给代理服务器
    client_socket.sendall(request_data)

    # 接收代理服务器的响应数据
    response = b""
    while True:
        data = client_socket.recv(1024)
        response += data
        if data:
            break

    # 关闭连接
    client_socket.close()

    return response

def build_request(hostname, url):
    # 构建GET请求报文
    request_lines = [
        f"GET {url} HTTP/1.1",
        f"Host: {hostname}",
        "User-Agent: Wget/1.21.4",
        "Accept: */*",
        "Accept-Encoding: identity",
        "Connection: Keep-Alive",
        "Proxy-Connection: Keep-Alive",
        "\r\n"
    ]
    request_data = "\r\n".join(request_lines).encode()
    return request_data

def main(proxy_host, proxy_port,url):
    # 解析代理服务器地址
    proxy_parts = urllib.parse.urlparse(f"http://{proxy_host}:{proxy_port}")
    proxy_hostname = proxy_parts.hostname
    proxy_port = proxy_parts.port

    # 构建请求报文
    

    hostname=urllib.parse.urlparse(url).hostname
    request_data = build_request(hostname, url)

    # 发送请求给代理服务器并接收响应
    response = send_request(proxy_hostname, proxy_port, request_data)

    # 打印响应内容
    print(response.decode())
