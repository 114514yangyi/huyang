#include <vector>
#include <cstdlib>

void swap(int& A, int& B) {
    int temp = A;
    A = B;
    B = temp;
}

int partition(std::vector<int>& nums, int L, int R) {
    int pivot = rand() % (R - L + 1) + L;
    swap(nums[pivot], nums[R]);
    int i = L - 1;
    for (int j = L; j < R; j++) {
        if (nums[j] < nums[R]) {
            i++;
            swap(nums[i], nums[j]);
        }
    }
    i++;
    swap(nums[i], nums[R]);
    return i;
}

int selection(std::vector<int>& nums, int index, int L, int R) {
    if (nums.size() == 0) return -1;
    if (L == R) return nums[L];
    int pivot = partition(nums, L, R);
    if (pivot == index) return nums[pivot];
    if (pivot > index) return selection(nums, index, L, pivot - 1);
    else return selection(nums, index, pivot + 1, R);
}
